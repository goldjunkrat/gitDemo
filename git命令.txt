使用git:
	git status
		查看当前仓库的状态
	git init
		初始化仓库
	git add 文件名
		将文件从未跟踪切换到暂存状态
	git add *
		将所有已修改(未跟踪)的文件暂存
	git commit -m "提交信息"
		将暂存的文件存储到仓库中,使文件从暂存切换到未修改状态
	git commit -a
		将所有已修改的文件直接提交到代码仓库中,省去add的流程(未跟踪的文件不会提交)
